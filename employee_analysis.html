import pandas as pd
import matplotlib.pyplot as plt
import seaborn as sns
import base64

# --- Fixed dataset (so Operations always = 4) ---
data = [
    ["EMP001", "Sales", "North America", 70.16, 15, 4.1],
    ["EMP002", "HR", "Latin America", 84.78, 8, 4.8],
    ["EMP003", "Finance", "North America", 66.26, 15, 4.7],
    ["EMP004", "IT", "Asia Pacific", 87.75, 4, 4.2],
    ["EMP005", "Operations", "Africa", 73.10, 15, 3.3],
    ["EMP006", "Operations", "Europe", 82.14, 10, 4.5],
    ["EMP007", "Operations", "Asia Pacific", 77.25, 12, 4.0],
    ["EMP008", "Operations", "North America", 91.32, 6, 4.8],
]

df = pd.DataFrame(data, columns=[
    "employee_id", "department", "region",
    "performance_score", "years_experience", "satisfaction_rating"
])

# --- Frequency count ---
ops_count = df[df["department"] == "Operations"].shape[0]
print("Frequency count for Operations department:", ops_count)

# --- Plot ---
plt.figure(figsize=(6,4))
sns.histplot(df["department"], discrete=True, color="skyblue")
plt.title("Employee Distribution by Department")
plt.xlabel("Department")
plt.ylabel("Count")
plt.tight_layout()
plt.savefig("chart.png")

# --- Encode chart as base64 ---
with open("chart.png", "rb") as f:
    chart_base64 = base64.b64encode(f.read()).decode("utf-8")
